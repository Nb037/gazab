function startSparkles() {
    setInterval(() => {
        const s = document.createElement("div");
        s.className = "sparkle";
        const size = Math.random() * 5 + 2 + "px";
        s.style.width = size;
        s.style.height = size;
        s.style.left = Math.random() * 100 + "vw";
        document.body.appendChild(s);
        setTimeout(() => s.remove(), 6000);
    }, 500);
}
